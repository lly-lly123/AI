## 🧪 测试脚本

### 快速测试脚本

在移动端页面控制台执行：

```javascript
// 移动端按钮点击测试
(function() {
  console.log('🔍 开始测试移动端按钮...\n');
  
  // 切换到更多功能视图
  if (typeof window.switchView === 'function') {
    window.switchView('more');
    
    setTimeout(() => {
      const moreView = document.getElementById('moreView');
      if (!moreView) {
        console.error('❌ 更多功能视图不存在');
        return;
      }
      
      const cards = moreView.querySelectorAll('.mobile-card');
      console.log(`找到 ${cards.length} 个功能卡片\n`);
      
      cards.forEach((card, index) => {
        const title = card.querySelector('.mobile-card-title')?.textContent?.trim() || `卡片${index + 1}`;
        const style = window.getComputedStyle(card);
        
        console.log(`卡片 ${index + 1}: ${title}`);
        console.log(`  pointer-events: ${style.pointerEvents}`);
        console.log(`  cursor: ${style.cursor}`);
        console.log(`  onclick: ${card.getAttribute('onclick') || '无'}`);
        
        // 检查修复函数
        if (typeof window.fixMoreViewCards === 'function') {
          console.log('  ✅ fixMoreViewCards函数存在');
        } else {
          console.log('  ❌ fixMoreViewCards函数不存在');
        }
        
        // 测试点击
        console.log('  测试点击...');
        const beforeView = window.getComputedStyle(moreView).display;
        card.click();
        setTimeout(() => {
          const afterView = window.getComputedStyle(moreView).display;
          if (afterView === 'none') {
            console.log('  ✅ 点击成功，视图已切换\n');
          } else {
            console.log('  ❌ 点击失败，视图未切换\n');
          }
        }, 200);
      });
    }, 500);
  }
})();
```
